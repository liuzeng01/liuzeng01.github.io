toml

TOML是前GitHub CEO， Tom Preston-Werner，于2013年创建的语言，其目标是成为一个小规模的易于使用的语义化配置文件格式。TOML被设计为可以无二义性的转换为一个哈希表(Hash table)。TOML是大小写敏感的，必须是UTF-8编码。

title = "TOML Example"

[owner]
name = "Lance Uppercut"
dob = 1979-05-27T07:32:00-08:00 # 日期是一等公民

[database]
server = "192.168.1.1"
ports = [ 8001, 8001, 8002 ]
connection_max = 5000
enabled = true

[servers]
  #你可以使用空格、制表符进行缩进，或者根本不缩进。TOML不关心缩进。
  [servers.alpha]
  ip = "10.0.0.1"
  dc = "eqdc10"

  [servers.beta]
  ip = "10.0.0.2"
  dc = "eqdc10"

[clients]
data = [ ["gamma", "delta"], [1, 2] ]

# 数组内可以混入换行符
hosts = [
  "alpha",
  "omega"
]

注释

注释使用#开头

基本字符串

字符串使用 " "包裹，所有Unicode字符均可出现，除了双引号、反斜线、控制字符(U+0000 to U+001F)需要转义。

多行字符串

多行字符串使用"""包裹，换行会进行保留

字面量字符串

使用' ' 包裹，内部不允许转义，例如 winpath = 'C:\Users\nodejs\templates'

多行-字面量字符串

使用''' ''' 进行包裹

日期

使用此种格式 date1 = 1979-05-27T07:32:00Z

数组

数组使用方括号包裹。空格会被忽略，包括换行符。元素使用逗号分隔。

arr1 = [ 1, 2, 3 ]
arr2 = [ "red", "yellow", "green" ]
arr3 = [ [ 1, 2 ], [3, 4, 5] ]

表格(注意概念)

表格叶称为哈希表或字典，用来存储键值对。表格名由方括号包裹，且自成一行。

[database]
server = "192.168.1.1"
ports = [ 8001, 8001, 8002 ]
connection_max = 5000
enabled = true

  #你可以使用空格、制表符进行缩进，或者根本不缩进。TOML不关心缩进。

  ip = "10.0.0.1"
  dc = "eqdc10"

  [servers.beta]
  ip = "10.0.0.2"
  dc = "eqdc10"
